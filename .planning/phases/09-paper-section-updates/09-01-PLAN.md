```
\paragraph{Solver and Scope.}
The MILP is solved using Gurobi~10 with a 300-second time limit; Gurobi reports
the optimality gap for instances where the time limit binds.

\textbf{Ex-post benchmark role.}
The MILP serves as an \emph{ex-post} benchmark, not a real-time solver.
Given the realized accepted set $\mathcal{R}^{\text{acc}}$ produced by a
completed ALNS run --- i.e., the set of requests for which the sampled
acceptance indicator $z_r = 1$ --- the MILP solves the optimal vehicle routing
for that \emph{fixed} accepted set.
Formally, the MILP treats $z_r$ as a known constant (equal to 1 for
$r \in \mathcal{R}^{\text{acc}}$ and 0 otherwise) and optimizes only over the
assignment variables $x_{r,m^P,m^D,v}$ and schedule variables $s_{v,i}$.
It does \emph{not} re-optimize the acceptance decisions: the stochastic
passenger responses are realized outcomes of the ALNS run, not MILP decision
variables.
The resulting MILP objective value is therefore a lower bound on the routing
cost achievable for the same accepted set, and the optimality gap
$\delta = (C^{\text{ALNS}} - C^{\text{MILP}}) / C^{\text{MILP}} \times 100\%$
measures how close the ALNS heuristic comes to the ex-post routing optimum.
Instances with up to 30--50 passengers and 5--8 vehicles are tractable within
the time limit.
```

Create the file with the following structure:

```python
"""
experiments/milp_gap.py — REV-10: MILP vs ALNS optimality gap experiment.

For small instances (n=20, n=30 passengers), runs:
  1. ALNS heuristic to get the realized accepted set and ALNS vehicle-km.
  2. MILP ex-post benchmark: given the realized accepted set from step 1,
     solve optimal vehicle routing for that fixed set.
  3. Compute optimality gap = (alns_vkm - milp_vkm) / milp_vkm * 100%.

Writes results to results/milp_gap.json.
"""
from __future__ import annotations
import json, os, time
from experiments.config import SEEDS, RHO_P, RHO_D, ALPHA_WEIGHTS, VEHICLE_CAPACITY, MAX_RIDE_TIME, MAX_ROUTE_DURATION
from experiments.scenarios import make_synthetic_scenario
from experiments.runner import run_variant   # returns metrics dict with 'vkm' key
from drt.types import Request, Vehicle, MeetingPoint


def run_gap_experiment(n_requests: int, seed: int) -> dict:
    """Run ALNS then MILP ex-post on the same instance. Return gap metrics."""
    n_vehicles = max(3, n_requests // 5)   # ~5 requests per vehicle for small instances
    requests, vehicles, meeting_points = make_synthetic_scenario(
        n_requests=n_requests, n_vehicles=n_vehicles, seed=seed
    )

    # Step 1: run ALNS (FullModel variant) to get realized accepted set + vkm
    alns_metrics = run_variant("FullModel", requests, vehicles, meeting_points)
    alns_vkm = alns_metrics["vkm"]
    accepted_ids = set(alns_metrics.get("accepted_ids", []))

    # Step 2: filter to realized accepted set (z_r = 1 from ALNS)
    accepted_requests = [r for r in requests if r.id in accepted_ids]
    if not accepted_requests:
        return {
            "n_requests": n_requests, "seed": seed,
            "alns_vkm": alns_vkm, "milp_vkm": None,
            "gap_pct": None, "milp_status": "no_accepted", "solve_time": 0.0,
        }

    # Step 3: MILP ex-post — optimize routing for fixed accepted set
    from drt.milp import DRTModel   # deferred import (requires Gurobi)
    milp = DRTModel(
        requests=accepted_requests,
        vehicles=vehicles,
        meeting_points=meeting_points,
        rho_p=RHO_P,
        rho_d=RHO_D,
        cost_weights=tuple(ALPHA_WEIGHTS),
        time_limit=300.0,
        mip_gap=0.01,
    )
    result = milp.solve()
    milp_vkm = result["objective_value"]
    gap_pct = None
    if result["status"] in ("optimal", "feasible") and milp_vkm and milp_vkm > 0:
        gap_pct = (alns_vkm - milp_vkm) / milp_vkm * 100.0

    return {
        "n_requests": n_requests, "seed": seed,
        "alns_vkm": alns_vkm, "milp_vkm": milp_vkm,
        "gap_pct": gap_pct, "milp_status": result["status"],
        "solve_time": result["solve_time"],
    }


def main():
    rows = []
    for n in [20, 30]:
        for seed in SEEDS:
            print(f"  n={n}, seed={seed} ...", flush=True)
            row = run_gap_experiment(n, seed)
            rows.append(row)
            print(f"    gap={row['gap_pct']:.1f}%" if row["gap_pct"] is not None else "    gap=N/A")

    os.makedirs("results", exist_ok=True)
    with open("results/milp_gap.json", "w") as f:
        json.dump(rows, f, indent=2)
    print("Wrote results/milp_gap.json")


if __name__ == "__main__":
    main()
```
