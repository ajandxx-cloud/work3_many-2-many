```latex
\subsection{Single-Offer Mechanism and Binary Logit Acceptance}

The system operates a \emph{single-offer mechanism}: Layer~1 selects the best feasible
bundle $b^* \in \mathcal{B}_r$ and presents \emph{only} $b^*$ to the passenger. The
passenger never observes the full set $\mathcal{B}_r$; they choose between accepting $b^*$
or taking the outside option (e.g., private car, walking, trip cancellation).

The utility of the outside option is normalized to zero:
\begin{equation}\label{eq:utility-outside}
  U_{r0}^k = 0
\end{equation}

Because the passenger's choice set contains exactly two alternatives --- $b^*$ and the
outside option --- the acceptance probability follows a \emph{binary logit} model:
\begin{equation}\label{eq:binary-logit}
  P_{\text{accept}}(b^*) = \frac{\exp(U_{rb^*}^k)}{\exp(U_{r0}^k) + \exp(U_{rb^*}^k)}
                         = \frac{\exp(U_{rb^*}^k)}{1 + \exp(U_{rb^*}^k)}
\end{equation}

The realized acceptance indicator is:
\begin{equation}\label{eq:acceptance-indicator}
  z_r \sim \text{Bernoulli}\!\left(P_{\text{accept}}(b^*)\right)
\end{equation}

This formulation is behaviorally consistent with the single-offer mechanism: the passenger
responds to the one bundle they are shown, not to a hypothetical full menu. The binary
logit is the special case of MNL with $|\text{choice set}| = 2$, so the standard
random-utility interpretation (i.i.d.\ Gumbel$(0,1)$ errors) is preserved.
```

```latex
\begin{algorithm}[h]
\caption{Rolling Horizon ALNS for Bidirectional DARPmp}
\label{alg:rh-alns}
\begin{algorithmic}[1]
\Require Request stream $\mathcal{R}$, fleet $\mathcal{V}$, meeting points $\mathcal{M}$,
         horizon $H$, re-opt interval $\Delta$, rejection penalty $\Gamma$
\Ensure  Accepted set $\mathcal{R}^{\text{acc}}$, routes $\{\sigma_v\}_{v \in \mathcal{V}}$

\State Initialize routes $\sigma_v \leftarrow \emptyset$ for all $v \in \mathcal{V}$
\State $t^{\text{now}} \leftarrow 0$

\While{request stream not exhausted}
  \State \textbf{// --- Online dispatch: process next arriving request ---}
  \State Receive request $r$ at time $t_r^{\text{req}}$
  \State $M_r^P \leftarrow \{m \in \mathcal{M} \mid d(o_r, m) \leq \rho^P\}$,
         $M_r^D \leftarrow \{m \in \mathcal{M} \mid d(\delta_r, m) \leq \rho^D\}$
         \Comment{Candidate meeting points}
  \State $b^* \leftarrow \textsc{BestInsertion}(r, M_r^P, M_r^D, \{\sigma_v\})$
         \Comment{Layer 1: select best feasible bundle}
  \If{$b^* = \emptyset$}
    \State $z_r \leftarrow 0$; record rejection
    \Comment{No feasible insertion found}
  \Else
    \State $p_{\text{acc}} \leftarrow \dfrac{\exp(U_{rb^*}^{k_r})}{1 + \exp(U_{rb^*}^{k_r})}$
           \Comment{Binary logit acceptance (Eq.~\ref{eq:binary-logit})}
    \State $z_r \sim \text{Bernoulli}(p_{\text{acc}})$
           \Comment{Layer 2: passenger response}
    \If{$z_r = 1$}
      \State Insert $b^*$ into $\sigma_{v_{b^*}}$ at positions $(\pi^P, \pi^D)$
      \State $\mathcal{R}^{\text{acc}} \leftarrow \mathcal{R}^{\text{acc}} \cup \{r\}$
    \EndIf
  \EndIf

  \State \textbf{// --- Periodic re-optimization ---}
  \If{$t^{\text{now}} \bmod \Delta = 0$}
    \State $\mathcal{R}^{\text{act}} \leftarrow \{r' \in \mathcal{R}^{\text{acc}} \mid r' \text{ not yet completed}\}$
    \State $\{\sigma_v\} \leftarrow \textsc{ALNS}(\mathcal{R}^{\text{act}}, \{\sigma_v\}, H)$
           \Comment{Layer 3: rolling horizon re-opt}
  \EndIf
  \State $t^{\text{now}} \leftarrow t^{\text{now}} + \delta t$
\EndWhile
\State \Return $\mathcal{R}^{\text{acc}},\; \{\sigma_v\}$
\end{algorithmic}
\end{algorithm}
```
