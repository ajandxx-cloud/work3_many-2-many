```python
import gurobipy as gp
from gurobipy import GRB
import math
from drt.types import Request, Vehicle, MeetingPoint
from drt.candidate import euclidean

class DRTModel:
    """
    EXACT-01: Static snapshot MILP for many-to-many DRT with bidirectional meeting points.
    Solves a batch of requests assigned to vehicles in a single optimization window.
    """
    def __init__(self, requests, vehicles, meeting_points,
                 rho_p, rho_d, cost_weights=(1,1,1,1,5),
                 travel_speed=1.0, time_limit=300.0, mip_gap=0.05):
        self.requests = requests          # list[Request]
        self.vehicles = vehicles          # list[Vehicle]
        self.meeting_points = meeting_points  # list[MeetingPoint]
        self.rho_p = rho_p
        self.rho_d = rho_d
        self.alpha = cost_weights         # (α1,α2,α3,α4,α5) for C^op,C^wait,C^walk,C^IVT,C^rej
        self.speed = travel_speed
        self.time_limit = time_limit
        self.mip_gap = mip_gap
        self.model = None
        self.result = None

    def _travel_time(self, a, b):
        return euclidean(a, b) / self.speed

    def build(self):
        """Build the Gurobi model. Call before solve()."""
        m = gp.Model("drt_milp")
        m.setParam("TimeLimit", self.time_limit)
        m.setParam("MIPGap", self.mip_gap)
        m.setParam("OutputFlag", 0)  # suppress Gurobi console output

        R = self.requests
        V = self.vehicles
        MP = self.meeting_points
        speed = self.speed

        # Candidate meeting points per request (filter by walking radius)
        from drt.candidate import generate_candidates
        cand_p = {r.id: generate_candidates(r, MP, self.rho_p, len(MP), 'pickup') for r in R}
        cand_d = {r.id: generate_candidates(r, MP, self.rho_d, len(MP), 'dropoff') for r in R}

        # --- Decision variables ---
        # z[r] = 1 if request r is accepted (EXACT-01)
        z = m.addVars([r.id for r in R], vtype=GRB.BINARY, name="z")

        # x[r,v,mp_p,mp_d] = 1 if request r assigned to vehicle v with pickup mp_p, dropoff mp_d
        x = {}
        for r in R:
            for v in V:
                for mp_p in cand_p[r.id]:
                    for mp_d in cand_d[r.id]:
                        x[r.id, v.id, mp_p.id, mp_d.id] = m.addVar(
                            vtype=GRB.BINARY,
                            name=f"x_{r.id}_{v.id}_{mp_p.id}_{mp_d.id}"
                        )

        # s[r,v,'p'] = scheduled pickup time; s[r,v,'d'] = scheduled dropoff time
        s = {}
        BIG_M = 1e6
        for r in R:
            for v in V:
                s[r.id, v.id, 'p'] = m.addVar(lb=0, ub=BIG_M, name=f"s_p_{r.id}_{v.id}")
                s[r.id, v.id, 'd'] = m.addVar(lb=0, ub=BIG_M, name=f"s_d_{r.id}_{v.id}")

        m.update()

        # --- Constraints ---

        # con:capacity — each vehicle serves at most capacity requests simultaneously
        # (simplified: total assigned requests per vehicle <= capacity)
        for v in V:
            m.addConstr(
                gp.quicksum(x[r.id, v.id, mp_p.id, mp_d.id]
                            for r in R
                            for mp_p in cand_p[r.id]
                            for mp_d in cand_d[r.id])
                <= v.capacity,
                name=f"con_capacity_{v.id}"
            )

        # con:assignment — each accepted request assigned to exactly one (vehicle, mp_p, mp_d)
        for r in R:
            m.addConstr(
                gp.quicksum(x[r.id, v.id, mp_p.id, mp_d.id]
                            for v in V
                            for mp_p in cand_p[r.id]
                            for mp_d in cand_d[r.id])
                == z[r.id],
                name=f"con_assign_{r.id}"
            )

        # con:tw-early — pickup time >= earliest (only binding when assigned)
        for r in R:
            for v in V:
                m.addConstr(
                    s[r.id, v.id, 'p'] >= r.earliest * gp.quicksum(
                        x[r.id, v.id, mp_p.id, mp_d.id]
                        for mp_p in cand_p[r.id] for mp_d in cand_d[r.id]
                    ),
                    name=f"con_tw_early_{r.id}_{v.id}"
                )

        # con:tw-late — pickup time <= latest (big-M formulation)
        for r in R:
            for v in V:
                assigned = gp.quicksum(
                    x[r.id, v.id, mp_p.id, mp_d.id]
                    for mp_p in cand_p[r.id] for mp_d in cand_d[r.id]
                )
                m.addConstr(
                    s[r.id, v.id, 'p'] <= r.latest + BIG_M * (1 - assigned),
                    name=f"con_tw_late_{r.id}_{v.id}"
                )

        # con:ridetime — dropoff - pickup <= max_ride_time
        for r in R:
            for v in V:
                m.addConstr(
                    s[r.id, v.id, 'd'] - s[r.id, v.id, 'p'] <= r.max_ride_time,
                    name=f"con_ridetime_{r.id}_{v.id}"
                )

        # con:precedence-time — pickup before dropoff
        for r in R:
            for v in V:
                m.addConstr(
                    s[r.id, v.id, 'd'] >= s[r.id, v.id, 'p'],
                    name=f"con_prec_time_{r.id}_{v.id}"
                )

        # con:walk-pickup — walking distance from origin to pickup mp <= rho_p
        # (enforced by candidate filtering; add explicit constraint for completeness)
        for r in R:
            for v in V:
                for mp_p in cand_p[r.id]:
                    for mp_d in cand_d[r.id]:
                        walk_p = euclidean(r.origin, mp_p.coords)
                        m.addConstr(
                            walk_p * x[r.id, v.id, mp_p.id, mp_d.id] <= self.rho_p,
                            name=f"con_walk_p_{r.id}_{v.id}_{mp_p.id}"
                        )

        # con:walk-dropoff — walking distance from dropoff mp to destination <= rho_d
        for r in R:
            for v in V:
                for mp_p in cand_p[r.id]:
                    for mp_d in cand_d[r.id]:
                        walk_d = euclidean(mp_d.coords, r.destination)
                        m.addConstr(
                            walk_d * x[r.id, v.id, mp_p.id, mp_d.id] <= self.rho_d,
                            name=f"con_walk_d_{r.id}_{v.id}_{mp_d.id}"
                        )

        # con:time-consistency — schedule time >= vehicle start time
        for r in R:
            for v in V:
                m.addConstr(
                    s[r.id, v.id, 'p'] >= v.current_time,
                    name=f"con_time_consist_{r.id}_{v.id}"
                )

        # con:route-duration — total route duration <= max_route_duration (big-M)
        for v in V:
            for r in R:
                m.addConstr(
                    s[r.id, v.id, 'd'] - v.current_time <= v.max_route_duration,
                    name=f"con_route_dur_{r.id}_{v.id}"
                )

        # --- Objective ---
        # min α1·C^op + α2·C^wait + α3·C^walk + α4·C^IVT + α5·C^rej
        a1, a2, a3, a4, a5 = self.alpha

        c_op = gp.quicksum(
            euclidean(mp_p.coords, mp_d.coords) / speed * x[r.id, v.id, mp_p.id, mp_d.id]
            for r in R for v in V
            for mp_p in cand_p[r.id] for mp_d in cand_d[r.id]
        )
        c_wait = gp.quicksum(
            s[r.id, v.id, 'p']
            for r in R for v in V
        )
        c_walk = gp.quicksum(
            (euclidean(r.origin, mp_p.coords) + euclidean(mp_d.coords, r.destination))
            * x[r.id, v.id, mp_p.id, mp_d.id]
            for r in R for v in V
            for mp_p in cand_p[r.id] for mp_d in cand_d[r.id]
        )
        c_ivt = gp.quicksum(
            (s[r.id, v.id, 'd'] - s[r.id, v.id, 'p'])
            for r in R for v in V
        )
        c_rej = gp.quicksum(1 - z[r.id] for r in R)

        m.setObjective(a1*c_op + a2*c_wait + a3*c_walk + a4*c_ivt + a5*c_rej,
                       GRB.MINIMIZE)

        self.model = m
        self._z = z
        self._x = x
        self._s = s
        self._cand_p = cand_p
        self._cand_d = cand_d

    def solve(self) -> dict:
        """
        EXACT-02/03: Solve the model. Returns result dict with:
        - status: 'optimal' | 'feasible' | 'infeasible' | 'timeout'
        - obj_value: float
        - mip_gap: float
        - solve_time: float (seconds)
        - accepted: list of request ids accepted
        """
        if self.model is None:
            self.build()
        self.model.optimize()
        m = self.model
        status_map = {
            GRB.OPTIMAL: 'optimal',
            GRB.SUBOPTIMAL: 'feasible',
            GRB.INFEASIBLE: 'infeasible',
            GRB.TIME_LIMIT: 'timeout',
        }
        status = status_map.get(m.Status, 'unknown')
        result = {
            'status': status,
            'obj_value': m.ObjVal if m.SolCount > 0 else None,
            'mip_gap': m.MIPGap if m.SolCount > 0 else None,
            'solve_time': m.Runtime,
            'accepted': [],
        }
        if m.SolCount > 0:
            result['accepted'] = [
                r.id for r in self.requests
                if self._z[r.id].X > 0.5
            ]
        self.result = result
        return result
```
