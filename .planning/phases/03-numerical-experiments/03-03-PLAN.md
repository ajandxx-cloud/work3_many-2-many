```python
import csv, os, time
import pandas as pd
from experiments.config import SCALES, SEEDS, VEHICLE_COUNTS, BEIJING_SCALE
from experiments.scenarios import generate_synthetic, generate_beijing
from experiments.variants import ALL_VARIANTS
from experiments.metrics import compute_metrics

RESULTS_DIR = os.path.join(os.path.dirname(__file__), '..', 'results')

def run_all_experiments(
    scales=None,
    seeds=None,
    beijing=True,
    results_dir=RESULTS_DIR,
):
    """
    Run all 6 variants across all scales and seeds.

    Args:
        scales: list of request counts (default: SCALES from config)
        seeds: list of random seeds (default: SEEDS from config)
        beijing: whether to run Beijing scenario (default: True)
        results_dir: output directory for CSV files
    """
    scales = scales or SCALES
    seeds = seeds or SEEDS
    os.makedirs(results_dir, exist_ok=True)

    synthetic_rows = []
    beijing_rows = []

    for scale in scales:
        n_vehicles = VEHICLE_COUNTS.get(scale, max(1, scale // 10))
        for seed in seeds:
            scenario = generate_synthetic(scale, n_vehicles, seed)
            for variant in ALL_VARIANTS:
                result = variant.run(scenario)
                m = compute_metrics(result)
                row = _metrics_row(variant.name, scale, seed, m)
                synthetic_rows.append(row)
                print(f"[synthetic] {variant.name} scale={scale} seed={seed} accept={m.acceptance_rate:.3f}")

    if beijing:
        n_vehicles = VEHICLE_COUNTS.get(BEIJING_SCALE, 15)
        for seed in seeds:
            scenario = generate_beijing(BEIJING_SCALE, n_vehicles, seed)
            for variant in ALL_VARIANTS:
                result = variant.run(scenario)
                m = compute_metrics(result)
                row = _metrics_row(variant.name, BEIJING_SCALE, seed, m)
                beijing_rows.append(row)
                print(f"[beijing] {variant.name} seed={seed} accept={m.acceptance_rate:.3f}")

    _write_csv(synthetic_rows, os.path.join(results_dir, 'synthetic_results.csv'))
    _write_csv(beijing_rows, os.path.join(results_dir, 'beijing_results.csv'))
    _write_metrics_table(synthetic_rows, os.path.join(results_dir, 'metrics_table.csv'))

    return synthetic_rows, beijing_rows


def _metrics_row(variant_name, scale, seed, m):
    return {
        'variant': variant_name, 'scale': scale, 'seed': seed,
        'acceptance_rate': m.acceptance_rate, 'vehicle_km': m.vehicle_km,
        'avg_wait': m.avg_wait, 'p95_wait': m.p95_wait,
        'avg_walk': m.avg_walk, 'avg_ivt': m.avg_ivt,
        'detour_ratio': m.detour_ratio, 'fairness_index': m.fairness_index,
        'cpu_time': m.cpu_time,
    }


def _write_csv(rows, path):
    if not rows:
        return
    with open(path, 'w', newline='') as f:
        writer = csv.DictWriter(f, fieldnames=list(rows[0].keys()))
        writer.writeheader()
        writer.writerows(rows)


def _write_metrics_table(rows, path):
    """Aggregate mean±std per variant across all scales and seeds."""
    if not rows:
        return
    df = pd.DataFrame(rows)
    metric_cols = ['acceptance_rate', 'vehicle_km', 'avg_wait', 'p95_wait',
                   'avg_walk', 'avg_ivt', 'detour_ratio', 'fairness_index', 'cpu_time']
    agg = df.groupby('variant')[metric_cols].agg(['mean', 'std']).reset_index()
    agg.columns = ['variant'] + [f'{m}_{s}' for m, s in agg.columns[1:]]
    agg.to_csv(path, index=False)


if __name__ == '__main__':
    run_all_experiments()
```
